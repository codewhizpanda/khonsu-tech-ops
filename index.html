Setup

Connect to Google Sheets for persistent reports.

1

Create Your Google Sheet

Go to sheets.google.com, create a blank sheet, name it Khonsu Tech Operations.

2

Add the Apps Script

Go to Extensions → Apps Script, paste below, then Deploy → New Deployment → Web App (access: Anyone).

// Khonsu Tech Operations — Apps Script Backend v2 // Normalized schema: Products, Inventory, Sales, Purchase_Orders, PO_Items, Promotions, Freebies, Settings const SS = SpreadsheetApp.getActiveSpreadsheet(); function doPost(e) { try { const d = JSON.parse(e.postData.contents); if (d.action === 'init') return initSheets(); if (d.action === 'logSale') return logSale(d); if (d.action === 'saveProducts') return saveProducts(d); if (d.action === 'saveInventory') return saveInventory(d); if (d.action === 'updateInventoryItems') return updateInventoryItems(d); if (d.action === 'savePO') return savePO(d); if (d.action === 'updatePOStatus') return updatePOStatus(d); if (d.action === 'savePromotions') return savePromotions(d); if (d.action === 'saveFreebies') return saveFreebies(d); if (d.action === 'saveSettings') return saveSettings(d); if (d.action === 'saveUnits') return saveUnits(d); if (d.action === 'replaceUnits') return replaceUnits(d); if (d.action === 'updateUnitStatus') return updateUnitStatus(d); if (d.action === 'verifyPin') return verifyPin(d); if (d.action === 'setPin') return setPin(d); return respond({error:'Unknown action'}); } catch(err) { return respond({error:err.toString()}); } } function doGet(e) { if (e.parameter.action === 'ping') return respond({status:'ok'}); if (e.parameter.action === 'getAllData') return getAllData(); return respond({status:'Khonsu Tech OPS v2 running'}); } // ─── INIT ──────────────────────────────────────────────────────────────────── function initSheets() { const tabs = { 'Products': ['ProductKey','Category','Name','RAM','Storage','Colors','UnitPrice','SRP','Status'], 'Inventory': ['ProductKey','Stock','ReorderPoint','LastUpdated'], 'Sales': ['SaleID','Date','Time','SO','BundleCode','ItemName','Variant','Color','Qty','UnitPrice','SRP','SoldPrice','PasaPrice','Discount','NetSales','Payment','SoldType','Promoter','Staff','IsAddon','IsPromotion','CustomerName','CustomerContact','CustomerEmail','IMEI'], 'Purchase_Orders': ['POID','Date','Supplier','Approver','Status'], 'PO_Items': ['POID','ItemName','Qty','Color'], 'Promotions': ['BundleID','Name','Price','MainProductKey','MainProductName','AddonProductKey','AddonProductName'], 'Freebies': ['MainProductKey','FreebieProductKey','MainProductName','FreebieName'], 'Settings': ['Key','Value'], 'IMEI_Units': ['IMEI','ProductKey','ProductName','Color','Status','DRNumber','ReceivedDate','SONumber','SoldDate','IsDummy'] }; Object.entries(tabs).forEach(([name, headers]) => { let sh = SS.getSheetByName(name); if (!sh) sh = SS.insertSheet(name); if (sh.getLastRow() === 0) { sh.appendRow(headers); sh.getRange(1,1,1,headers.length).setFontWeight('bold').setBackground('#1b2e6b').setFontColor('white'); } }); return respond({status:'Initialized'}); } // ─── GET ALL DATA ───────────────────────────────────────────────────────────── function getAllData() { const result = {}; result.products = sheetToObjects('Products'); result.inventory = sheetToObjects('Inventory'); result.promotions = sheetToObjects('Promotions'); result.freebies = sheetToObjects('Freebies'); result.purchaseOrders = getPOs(); const raw = sheetToObjects('Settings'); result.settings = {}; raw.forEach(r => { if (r.Key) result.settings[r.Key] = r.Value; }); return respond(result); } function sheetToObjects(name) { const sh = SS.getSheetByName(name); if (!sh || sh.getLastRow() < 2) return []; const data = sh.getDataRange().getValues(); const h = data[0]; return data.slice(1).map(row => { const o = {}; h.forEach((k,i) => o[k]=row[i]); return o; }); } function getPOs() { const pos = sheetToObjects('Purchase_Orders'); const items = sheetToObjects('PO_Items'); const map = {}; pos.forEach(po => { po.items = []; map[po.POID] = po; }); items.forEach(item => { if (map[item.POID]) map[item.POID].items.push({name:item.ItemName, qty:Number(item.Qty)||0, color:item.Color}); }); return pos; } // ─── SALES ─────────────────────────────────────────────────────────────────── function logSale(d) { let sh = SS.getSheetByName('Sales'); if (!sh) return respond({error:'No Sales sheet — run Init first'}); const dt = new Date(d.date || Date.now()); const dateStr = dt.toLocaleDateString('en-PH'); const timeStr = dt.toLocaleTimeString('en-PH'); (d.rows||[]).forEach(r => { sh.appendRow([r.id||'', dateStr, timeStr, r.so||'', r.bundle||'', r.itemName, r.variant||'', r.color||'', r.qty, r.unitPrice||0, r.srp, r.soldPrice, r.pasaPrice||0, r.discount||0, r.netSales||0, r.payment, r.soldType, r.promoter||'', r.staff, r.isAddon?'Yes':'', r.isPromotion?'Yes':'', r.customer?(r.customer.name||''):'', r.customer?(r.customer.contact||''):'', r.customer?(r.customer.email||''):'', (r.imeis||[]).join(', ') ]); }); return respond({status:'Sale logged', count:(d.rows||[]).length}); } // ─── PRODUCTS ──────────────────────────────────────────────────────────────── function saveProducts(d) { const sh = getOrCreateSheet('Products', ['ProductKey','Category','Name','RAM','Storage','Colors','UnitPrice','SRP','Status']); clearData(sh); (d.rows||[]).forEach(r => sh.appendRow(r)); return respond({status:'Products saved', count:(d.rows||[]).length}); } // ─── INVENTORY ─────────────────────────────────────────────────────────────── function saveInventory(d) { const sh = getOrCreateSheet('Inventory', ['ProductKey','Stock','ReorderPoint','LastUpdated']); clearData(sh); const now = new Date().toLocaleString('en-PH'); (d.rows||[]).forEach(r => sh.appendRow([r.productKey, r.stock, r.reorder||1, now])); return respond({status:'Inventory saved'}); } function updateInventoryItems(d) { const sh = SS.getSheetByName('Inventory'); if (!sh) return respond({error:'No Inventory sheet'}); const data = sh.getDataRange().getValues(); const now = new Date().toLocaleString('en-PH'); (d.items||[]).forEach(item => { let found = false; for (let i = 1; i < data.length; i++) { if (String(data[i][0]).trim() === String(item.productKey).trim()) { sh.getRange(i+1,2).setValue(item.stock); sh.getRange(i+1,4).setValue(now); found = true; break; } } if (!found) sh.appendRow([item.productKey, item.stock, item.reorder||1, now]); }); return respond({status:'ok'}); } // ─── PURCHASE ORDERS ───────────────────────────────────────────────────────── function savePO(d) { const posh = getOrCreateSheet('Purchase_Orders', ['POID','Date','Supplier','Approver','Status']); const poish = getOrCreateSheet('PO_Items', ['POID','ItemName','Qty','Color']); const poData = posh.getDataRange().getValues(); let poRow = -1; for (let i = 1; i < poData.length; i++) { if (String(poData[i][0]) === String(d.id)) { poRow = i + 1; break; } } const row = [d.id, d.date, d.supplier, d.approver, d.status||'pending']; if (poRow > 0) posh.getRange(poRow,1,1,5).setValues([row]); else posh.appendRow(row); const iData = poish.getDataRange().getValues(); const toDelete = []; for (let i = iData.length - 1; i >= 1; i--) { if (String(iData[i][0]) === String(d.id)) toDelete.push(i + 1); } toDelete.forEach(r => poish.deleteRow(r)); (d.items||[]).forEach(item => poish.appendRow([d.id, item.name, item.qty, item.color||'Assorted'])); return respond({status:'PO saved'}); } function updatePOStatus(d) { const sh = SS.getSheetByName('Purchase_Orders'); if (!sh) return respond({error:'No Purchase_Orders sheet'}); const data = sh.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (String(data[i][0]) === String(d.id)) { sh.getRange(i+1,5).setValue(d.status); return respond({status:'ok'}); } } return respond({error:'PO not found'}); } // ─── PROMOTIONS ────────────────────────────────────────────────────────────── function savePromotions(d) { const sh = getOrCreateSheet('Promotions', ['BundleID','Name','Price','MainProductKey','MainProductName','AddonProductKey','AddonProductName']); clearData(sh); (d.bundles||[]).forEach(b => sh.appendRow([b.id, b.name, b.price, b.mainKey, b.mainName, b.addonKey, b.addonName])); return respond({status:'Promotions saved'}); } // ─── FREEBIES ──────────────────────────────────────────────────────────────── function saveFreebies(d) { const sh = getOrCreateSheet('Freebies', ['MainProductKey','FreebieProductKey','MainProductName','FreebieName']); clearData(sh); (d.freebies||[]).forEach(f => sh.appendRow([f.mainKey, f.freebieKey, f.mainName||'', f.freebieName||''])); return respond({status:'Freebies saved'}); } // ─── SETTINGS ──────────────────────────────────────────────────────────────── function saveSettings(d) { const sh = getOrCreateSheet('Settings', ['Key','Value']); clearData(sh); Object.entries(d.settings||{}).forEach(([k,v]) => sh.appendRow([k,v])); return respond({status:'Settings saved'}); } // ─── IMEI UNITS ────────────────────────────────────────────────────────────── function saveUnits(d) { const sh = getOrCreateSheet('IMEI_Units', ['IMEI','ProductKey','ProductName','Color','Status','DRNumber','ReceivedDate','SONumber','SoldDate','IsDummy']); (d.units||[]).forEach(u => sh.appendRow([u.imei, u.productKey, u.productName, u.color, u.status||'available', u.drNumber||'', u.receivedDate||'', u.soNumber||'', u.soldDate||'', u.isDummy?'Yes':'No'])); return respond({status:'Units saved', count:(d.units||[]).length}); } function replaceUnits(d) { const sh = getOrCreateSheet('IMEI_Units', ['IMEI','ProductKey','ProductName','Color','Status','DRNumber','ReceivedDate','SONumber','SoldDate','IsDummy']); clearData(sh); (d.units||[]).forEach(u => sh.appendRow([u.imei, u.productKey, u.productName, u.color, u.status||'available', u.drNumber||'', u.receivedDate||'', u.soNumber||'', u.soldDate||'', u.isDummy?'Yes':'No'])); return respond({status:'Units replaced', count:(d.units||[]).length}); } function updateUnitStatus(d) { const sh = SS.getSheetByName('IMEI_Units'); if (!sh) return respond({error:'No IMEI_Units sheet — run Init first'}); const data = sh.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (String(data[i][0]) === String(d.imei)) { sh.getRange(i+1,5).setValue('sold'); sh.getRange(i+1,8).setValue(d.soNumber||''); sh.getRange(i+1,9).setValue(d.soldDate||new Date().toLocaleDateString('en-PH')); return respond({status:'ok'}); } } return respond({error:'IMEI not found'}); } // ─── ADMIN PIN ─────────────────────────────────────────────────────────────── function pinHash(pin) { const bytes = Utilities.computeDigest(Utilities.DigestAlgorithm.SHA_256, String(pin), Utilities.Charset.UTF_8); return bytes.map(b => ('0' + (b & 0xFF).toString(16)).slice(-2)).join(''); } function getStoredPinHash() { const props = PropertiesService.getScriptProperties(); let h = props.getProperty('ADMIN_PIN_HASH'); if (!h) { h = pinHash('1234'); props.setProperty('ADMIN_PIN_HASH', h); } return h; } function verifyPin(d) { return respond({valid: pinHash(d.pin||'') === getStoredPinHash()}); } function setPin(d) { if (pinHash(d.current||'') !== getStoredPinHash()) return respond({error:'Incorrect current PIN'}); const next = String(d.next||''); if (next.length < 4) return respond({error:'PIN must be at least 4 digits'}); PropertiesService.getScriptProperties().setProperty('ADMIN_PIN_HASH', pinHash(next)); return respond({status:'ok'}); } // ─── HELPERS ───────────────────────────────────────────────────────────────── function respond(data) { return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON); } function getOrCreateSheet(name, headers) { let sh = SS.getSheetByName(name); if (!sh) { sh = SS.insertSheet(name); sh.appendRow(headers); sh.getRange(1,1,1,headers.length).setFontWeight('bold').setBackground('#1b2e6b').setFontColor('white'); } return sh; } function clearData(sh) { if (sh.getLastRow() > 1) sh.deleteRows(2, sh.getLastRow() - 1); }
